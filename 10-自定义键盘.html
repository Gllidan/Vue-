window.onload = function () {

        Vue.directive('on').keyCodes.control=17;
        new Vue({
            el:'body',
            data:{
                age:'19'
            },
            methods:{
                show:function (ev) {
                    alert(ev.keyCode);
                }
            }
        })

    };

<!--可以把17自定义成为ctrol键-->
<input type="text" v-model="age" @keydown.control="show">
