window.onload = function () {

        new Vue({
           el:'body',
            data:{
               dataList:['width','height','background','orange'],
                numList:[1,2,3,4,5,6,7]
            },
            methods:{
               show:function () {
                   alert(1);
               }
            }
        });

    };
<!--在vue当中使用过滤器和angular当中非常一样.通过管道符的写法-->
<p>{{'hello' | uppercase}}</p>

<!--同样它也支持连写的形式-->
<p>{{'HELLO' | lowercase | capitalize}}</p>

<!--转成货币 注意它的传形式与angular有些区别, 它不要: 直接在后面空格写上参数-->
<p>{{12|currency '@'}}</p>

<!--过滤数组 filterBy '过滤条件' -->
<ul>
    <li v-for="val in dataList | filterBy 'a' ">{{val}}</li>
</ul>

<!--过滤数组 limitBy '开始位置' -->
<ul>
    <li v-for="val in dataList | limitBy 1">{{val}}</li>
</ul>

<!--排序 大于等于0为升序 小于为降序-->
<ul>
    <li v-for="val in numList | orderBy -1">{{val}}</li>
</ul>

<!--延时1秒后再去执行对应的方法-->
<input type="text" @keyup="show | debounce 1000"></input>
