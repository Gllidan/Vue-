<div @click="show">
        <!--在定义事件时, 可以传入事件对象,通过事件对象可以阻止事件的默认行为.-->
        <p @click="show($event)"> p标签 </p>
    </div>

    <!--坚听键盘事件-->
    <input type="text" @keydown="keydown($event)">
    <!--可以使用@keydown.enter/left/up/right/...监听各按键的点击-->
    <input type="text" @keydown.13="enter($event)">
    <input type="text" @keydown.enter="enter($event)">
    <input type="text" @keydown.left="enter($event)">

    new Vue({
        el:'body',
        data:{
            msg:"xmg"
        },
        /*所有的事件都定义在method方法当中
         * 在method中可以通过this来获取模型当中的内容
         * */
        methods:{
            show:function (e) {
                /*阻止事件默认行为*/
                e.cancelBubble=true;
                e.preventDefault();
                alert(1);
            },
            keydown:function (ev) {
                if (ev.keyCode == 13){
                    alert('按了回车');
                }

            },
            enter:function () {
                alert('按了回车');
            }
        }
    });
