<script>
    window.onload = function () {

        new Vue({

            el:'body',
            data:{
                myData:[],
                username:'',
                age:'',
                deleteTitle:'确认删除该条信息嘛',
                newIndex:-1
            },
            methods:{

                add:function () {
                    var user = {
                        userName:this.username,
                        age:this.age
                    }
                    this.myData.push(user);
                    this.reset();
                },
                reset:function () {
                    this.username = '';
                    this.age = '';
                },
                delete:function () {

                    if (this.newIndex == -2){
                        this.myData = [];
                    }else {
                        this.myData.splice(this.newIndex,1);
                    }

                },

                deleteAll:function () {
                    this.newIndex = -2;
                    this.deleteTitle = "确认删除全部嘛?";
                }


            }

        });


    };
</script>
